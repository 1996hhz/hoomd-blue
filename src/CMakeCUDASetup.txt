## Find CUDA
## This funny construction only allows the user to enable CUDA 
## if single precicsion is enabled
remove_definitions (-DSINGLE_PRECISION)
remove_definitions (-DUSE_CUDA)
if(SINGLE_PRECISION)
	add_definitions (-DSINGLE_PRECISION)

	option(USE_CUDA "Enable the compilation of the CUDA GPU code" off)

	if (USE_CUDA)
		add_definitions (-DUSE_CUDA)
		find_package(Cuda REQUIRED)
	endif (USE_CUDA)
