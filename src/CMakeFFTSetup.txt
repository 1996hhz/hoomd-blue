## Optional include of a FFT
OPTION(USE_FFT "A Fast Fourier Transform routine is defined" OFF)
if(USE_FFT)
add_definitions(-DUSE_FFT)
endif(USE_FFT)

if (USE_FFT)
option(USE_FFTW "Link to the FFTW library" OFF)
option(I_WILL_NOT_DISTRUBUTE_THIS_HOOMD "Enable to indicate that you will not distrubute this build of HOOMD" OFF)

if (USE_FFTW)
	if (NOT I_WILL_NOT_DISTRUBUTE_THIS_HOOMD)
		message(SEND_ERROR "FFTW is under the GPLv2 license. A HOOMD build linked to it cannot be distributed without  violating that license, as HOOMD is released under the new BSD license. Specify that you will not distribute this build of HOOMD by setting the I_WILL_NOT_DISTRUBUTE_THIS_HOOMD option to ON")
	endif (NOT I_WILL_NOT_DISTRUBUTE_THIS_HOOMD)
	
	# setup the FFTW search directories based on a FFTW root dir specified in CMake
	set(FFTW_ROOT "" CACHE FILEPATH "Root directory containing FFTW")
	set(FFTW_INCLUDE_SEARCH_DIRS
		${FFTW_ROOT}
		${FFTW_ROOT}/include
		)
	set(FFTW_LIBRARY_SEARCH_DIRS
		${FFTW_ROOT}
		${FFTW_ROOT}/lib64
		${FFTW_ROOT}/lib
		)
		
	# find the include directories for FFTW
	find_path(FFTW_INCLUDE_DIR
		fftw3.h
		HINTS ${FFTW_INCLUDE_SEARCH_DIRS}
		DOC "Inlclude directory for FFTW"
		)
		
	include_directories(${FFTW_INCLUDE_DIR})
		
	# find the fftw libraries
	find_library(FFTW_LIBRARY
		NAMES fftw3 fftw3-3
		HINTS ${FFTW_LIBRARY_SEARCH_DIRS}
		DOC "FFTW3 library"
		)
endif (USE_FFTW)
		

endif (USE_FFT)
